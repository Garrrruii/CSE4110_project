CREATE TABLE customer (  customer_ID char(10) NOT NULL,  customer_name varchar(20) NOT NULL,  customer_birth char(10) NOT NULL,  customer_phone varchar(13) NOT NULL,  customer_address varchar(100) NOT NULL,  PRIMARY KEY (customer_ID));
CREATE TABLE contract (  contract_account varchar(20) NOT NULL,  customer_ID char(10) NOT NULL,  PRIMARY KEY (contract_account),  KEY cont_cus_idx (customer_ID),  CONSTRAINT cont_cus FOREIGN KEY (customer_ID) REFERENCES customer (customer_ID));
CREATE TABLE package (  package_code char(8) NOT NULL,  package_detail varchar(25) NOT NULL,  package_type varchar(20) NOT NULL,  package_weight int NOT NULL,  package_timeline char(16) DEFAULT NULL,  package_hazard varchar(8) NOT NULL,  customer_ID char(10) DEFAULT NULL,  PRIMARY KEY (package_code),  KEY pack_cus_idx (customer_ID),  CONSTRAINT pack_cus FOREIGN KEY (customer_ID) REFERENCES customer (customer_ID));
CREATE TABLE receipt (  receipt_code char(8) NOT NULL,  receipt_type varchar(6) NOT NULL,  receipt_from varchar(100) NOT NULL,  receipt_to varchar(100) NOT NULL,  package_code char(8) NOT NULL,  recipient varchar(20) DEFAULT NULL,  PRIMARY KEY (receipt_code),  KEY rec_pack_idx (package_code),  CONSTRAINT rec_pack FOREIGN KEY (package_code) REFERENCES package (package_code));
CREATE TABLE bill (  bill_num char(8) NOT NULL,  bill_amount int NOT NULL,  bill_payment varchar(11) NOT NULL,  bill_state varchar(8) NOT NULL,  bill_issuetime char(16) NOT NULL,  receipt_code char(8) NOT NULL,  PRIMARY KEY (bill_num),  KEY bill_rec_idx (receipt_code),  CONSTRAINT bill_rec FOREIGN KEY (receipt_code) REFERENCES receipt (receipt_code));
CREATE TABLE shipment (  shipment_code char(8) NOT NULL,  shipment_type varchar(5) NOT NULL,  shipment_num varchar(6) NOT NULL,  PRIMARY KEY (shipment_code));
CREATE TABLE ship_pack (  shipment_code char(8) NOT NULL,  package_code char(8) NOT NULL,  PRIMARY KEY (shipment_code,package_code),  KEY ship_pack2_idx (package_code),  CONSTRAINT ship_pack1 FOREIGN KEY (shipment_code) REFERENCES shipment (shipment_code),  CONSTRAINT ship_pack2 FOREIGN KEY (package_code) REFERENCES package (package_code));
CREATE TABLE warehouse (  warehouse_code char(6) NOT NULL,  warehouse_name varchar(20) NOT NULL,  warehouse_address varchar(100) NOT NULL,  warehouse_phone varchar(13) NOT NULL,  PRIMARY KEY (warehouse_code));
CREATE TABLE delivery (  shipment_code char(8) NOT NULL,  delivery_time char(16) NOT NULL,  warehouse_to char(6) NOT NULL,  warehouse_from char(6) NOT NULL,  delivery_state varchar(6) NOT NULL,  PRIMARY KEY (shipment_code,delivery_time),  KEY del_to_idx (warehouse_to),  KEY del_from_idx (warehouse_from),  CONSTRAINT del_from FOREIGN KEY (warehouse_from) REFERENCES warehouse (warehouse_code),  CONSTRAINT del_ship FOREIGN KEY (shipment_code) REFERENCES shipment (shipment_code),  CONSTRAINT del_to FOREIGN KEY (warehouse_to) REFERENCES warehouse (warehouse_code));
CREATE TABLE staff (  staff_ID char(9) NOT NULL,  staff_job varchar(20) DEFAULT NULL,  staff_phone varchar(13) NOT NULL,  staff_birth char(10) NOT NULL,  staff_employ char(10) NOT NULL,  warehouse_code char(6) DEFAULT NULL,  staff_name varchar(20) NOT NULL,  PRIMARY KEY (staff_ID),  KEY work_idx (warehouse_code),  CONSTRAINT work FOREIGN KEY (warehouse_code) REFERENCES warehouse (warehouse_code));
CREATE TABLE manage (  shipment_code char(8) NOT NULL,  staff_ID char(9) NOT NULL,  PRIMARY KEY (shipment_code,staff_ID),  KEY manage_staff_idx (staff_ID),  CONSTRAINT manage_ship FOREIGN KEY (shipment_code) REFERENCES shipment (shipment_code),  CONSTRAINT manage_staff FOREIGN KEY (staff_ID) REFERENCES staff (staff_ID));
